package frc.robot.subsystems;

import edu.wpi.first.wpilibj.smartdashboard.SmartDashboard;
import edu.wpi.first.wpilibj.DigitalInput;
import edu.wpi.first.wpilibj2.command.SubsystemBase;

public class SensorSubsystem extends SubsystemBase {
  private DigitalInput upLimitSwitch;
  private DigitalInput downLimitSwitch;
  private DigitalInput collectSwitch;
  private DigitalInput tecniLimitSwitch;

  public void sensorDisplay(int upLimitSwitchPort, int downLimitSwitchPort, int collectSwitchPort, int tecniLimitSwitchPort) {
    this.upLimitSwitch = new DigitalInput(upLimitSwitchPort);
    this.downLimitSwitch = new DigitalInput(downLimitSwitchPort);
    this.collectSwitch = new DigitalInput(collectSwitchPort);
    this.tecniLimitSwitch = new DigitalInput(tecniLimitSwitchPort);
  }

  @Override
  public void periodic() {
    SmartDashboard.putBoolean("COLETA", collectSwitch.get());
    SmartDashboard.putBoolean("TECNIL", tecniLimitSwitch.get());
    SmartDashboard.putBoolean("ERGUIDO", upLimitSwitch.get());
    SmartDashboard.putBoolean("ABAIXADO", downLimitSwitch.get());
  }
}
